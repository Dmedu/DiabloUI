# 文本

## 用法

```
import { EZText } from 'react-native-diablo-ui';
const App = () => {
    return (
        <EZText h1 style={{color:'cyan'}}>
            title
            <EZText h4 h4Style={{,color:'#fff',fontSize:28}}>
                subTitle
            </EZText>
        </EZText>
    )
}
export default App;

```
